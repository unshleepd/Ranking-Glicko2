glicko

Aplikacja do zarządzania rankingami zawodników w systemie Glicko-2 z interfejsem graficznym (GUI) opartym na bibliotece Tkinter. Umożliwia dodawanie zawodników, rozgrywanie meczów, automatyczną aktualizację rankingów, wyświetlanie historii meczów, tworzenie wykresów zmian rankingów oraz eksportowanie danych.

 38    def __init__(self, nazwa, rating=None, rd=None, vol=None):
 39        """
 40        Inicjalizuje nowego zawodnika.
 41
 42        Args:
 43            nazwa (str): Nazwa zawodnika.
 44            rating (float, optional): Początkowy rating zawodnika. Jeśli None,

 46            rd (float, optional): Początkowa wartość RD zawodnika. Jeśli None,
 47                używany jest domyślny.
 48            vol (float, optional): Początkowa wartość volatility zawodnika.
 49                Jeśli None, używany jest domyślny.
 50        """
 51        self.nazwa = nazwa
 52        self.player = Player()
 53        if rating is not None:
 54            self.player.rating = rating
 55        if rd is not None:
 56            self.player.rd = rd
 57        if vol is not None:
 58            self.player.vol = vol
 59        self.oponenci = []
 60        self.wygrane = 0
 61        self.przegrane = 0
 62        self.remisy = 0
 63        self.historia_ratingow = [self.player.rating]

 76        """
 77        Przygotowuje stan obiektu do zapisu (serializacji).
 78
 79        Returns:
 80            dict: Słownik zawierający stan obiektu.

 82        state = {
 83            "nazwa": self.nazwa,
 84            "rating": self.player.rating,
 85            "rd": self.player.rd,
 86            "vol": self.player.vol,
 87            "wygrane": self.wygrane,
 88            "przegrane": self.przegrane,
 89            "remisy": self.remisy,
 90            "historia_ratingow": self.historia_ratingow,
 91        }

 97        Odtwarza obiekt Zawodnik ze stanu (deserializacja).

392    def aktualizuj_rankingi(self):
393        """
394        Aktualizuje rankingi wszystkich zawodników na podstawie rozegranych meczów.
395        """
396        for zawodnik in self.zawodnicy.values():
397            if zawodnik.oponenci:
398                rating_list = []  # Lista ratingów przeciwników
399                RD_list = []  # Lista RD przeciwników
400                outcome_list = []  # Lista wyników meczów
401                for opponent_player, wynik in zawodnik.oponenci:
402                    rating_list.append(opponent_player.rating)
403                    RD_list.append(opponent_player.rd)
404                    outcome_list.append(wynik)
405                # Aktualizacja rankingu zawodnika
406                zawodnik.player.update_player(
407                    rating_list, RD_list, outcome_list
408                )
409                # Dodanie nowego ratingu do historii
410                zawodnik.historia_ratingow.append(zawodnik.player.rating)
411                zawodnik.oponenci = []  # Wyczyść listę oponentów po aktualizacji
412        # Rankingi są aktualizowane automatycznie po każdym meczu

414    def pokaz_rankingi(self):
415        """
416        Wyświetla rankingi zawodników w polu tekstowym.
417        """
418        self.text_output.delete(1.0, tk.END)  # Wyczyść pole tekstowe
419        sort_key = self.var_sort.get()  # Pobierz wybrane kryterium sortowania
420
421        zawodnicy_list = list(self.zawodnicy.values())
422
423        try:
424            # Sortowanie listy zawodników według wybranego kryterium
425            if sort_key == "Rating":
426                zawodnicy_list.sort(
427                    key=lambda x: x.player.rating, reverse=True
428                )
429            elif sort_key == "Procent wygranych":
430                zawodnicy_list.sort(
431                    key=lambda x: (x.wygrane / (x.wygrane + x.przegrane + x.remisy))
432                    if (x.wygrane + x.przegrane + x.remisy) > 0
433                    else 0,
434                    reverse=True,
435                )
436            elif sort_key == "Liczba meczów":
437                zawodnicy_list.sort(
438                    key=lambda x: x.wygrane + x.przegrane + x.remisy,
439                    reverse=True,
440                )
441        except Exception as e:
442            messagebox.showerror("Błąd", f"Błąd podczas sortowania: {e}")
443            return
444
445        # Wyświetlenie informacji o każdym zawodniku
446        for zawodnik in zawodnicy_list:
447            rating = zawodnik.player.rating
448            rd = zawodnik.player.rd
449            vol = zawodnik.player.vol
450            mecze = zawodnik.wygrane + zawodnik.przegrane + zawodnik.remisy
451            if mecze > 0:
452                procent_wygranych = (zawodnik.wygrane / mecze) * 100
453            else:
454                procent_wygranych = 0
455            info = (
456                f"{zawodnik.nazwa}:\n"
457                f"  Rating: {rating:.2f}, RD: {rd:.2f}, Volatility: {vol:.5f}\n"
458                f"  Mecze: {mecze}, Wygrane: {zawodnik.wygrane}, "
459                f"Przegrane: {zawodnik.przegrane}, Remisy: {zawodnik.remisy}, "
460                f"Procent wygranych: {procent_wygranych:.2f}%\n\n"
461            )
462            self.text_output.insert(tk.END, info)
